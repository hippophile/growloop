# Learning Path

Reads skill gaps from memory, picks ONE focus for the week, delivers a practical micro-lesson in Obsidian.

## Generate this week's lesson

```bash
python3 << 'PYEOF'
import subprocess, os
from pathlib import Path
from datetime import datetime

env = os.environ.copy()
env.setdefault('DISPLAY', ':0')
try:
    dbus = subprocess.check_output(
        ['bash','-c','cat /proc/$(pgrep -u $USER gnome-session | head -1)/environ 2>/dev/null | tr "\\0" "\\n" | grep DBUS_SESSION_BUS_ADDRESS | cut -d= -f2-'],
        text=True).strip()
    if dbus: env['DBUS_SESSION_BUS_ADDRESS'] = dbus
except: pass

memory_dir = Path("__MEMORY_PATH__")
profile = (memory_dir/'user_profile.md').read_text() if (memory_dir/'user_profile.md').exists() else ''
workflows = (memory_dir/'project_workflows.md').read_text() if (memory_dir/'project_workflows.md').exists() else ''
lesson_log = memory_dir / 'learning_log.md'
past_lessons = lesson_log.read_text() if lesson_log.exists() else 'None yet.'

week = datetime.now().strftime('%Y-W%W')
report_date = datetime.now().strftime('%Y-%m-%d')

prompt = f"""You are a senior developer coaching a developer.
Pick ONE skill gap to focus on this week. Practical and immediately useful.

User profile:
{profile}

Common workflows:
{workflows}

Past lessons (don't repeat):
{past_lessons}

Return an Obsidian markdown note:

---
date: {report_date}
tags: [learning, weekly]
week: {week}
topic: [topic name]
---

# This Week: [topic]

## Why this matters to you
## The one concept
## The one command or action to try today
## How you'll know it worked

## Connected notes
- [[report-{report_date}]] — This week's dev report
- [[Learning Path]] — All lessons

---
*Generated by growloop — {report_date}*

SHORT. One thing done well beats ten skimmed."""

result = subprocess.run(
    ['claude', '--model', 'claude-haiku-4-5', '--print', prompt],
    capture_output=True, text=True, env=env
)

if result.returncode == 0:
    lesson = result.stdout.strip()

    with open(lesson_log, 'a') as f:
        f.write(f"\n## {week}\n{lesson[:200]}\n")

    vault = Path("__VAULT_PATH__")
    lesson_dir = vault / 'Claude/Learning Path'
    lesson_dir.mkdir(parents=True, exist_ok=True)
    lesson_path = lesson_dir / f'lesson-{week}.md'
    lesson_path.write_text(lesson + '\n')

    vault_name = vault.name
    subprocess.run([
        'zenity', '--info', '--title', "This Week's Focus",
        '--text', lesson[:600] + '\n\n(Full lesson in Obsidian → Claude/Learning Path)',
        '--ok-label', 'Open in Obsidian', '--width', '540'
    ], env=env)
    subprocess.run(['obsidian',
        f'obsidian://open?vault={vault_name.replace(" ", "%20")}&file=Claude/Learning%20Path/lesson-{week}'],
        env=env)
    print(lesson)
PYEOF
```
